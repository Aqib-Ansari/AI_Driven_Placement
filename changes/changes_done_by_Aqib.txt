change : SQL tables:
            1) student_register
                CREATE TABLE student_register (
                        id INT AUTO_INCREMENT PRIMARY KEY,
                        username VARCHAR(255),
                        email VARCHAR(255) UNIQUE,
                        password VARCHAR(255),
                        enrollment_num VARCHAR(255) UNIQUE,
                        college VARCHAR(255),
                        course VARCHAR(255),
                        year VARCHAR(4),
                        rollno INT
                    );

            2) resume 
                CREATE TABLE IF NOT EXISTS resume (
                    id INT AUTO_INCREMENT PRIMARY KEY,
                    filename VARCHAR(255),
                    pdf_data LONGBLOB
                )

            3) quiz_questions
                CREATE TABLE quiz_questions (
                    id INT AUTO_INCREMENT PRIMARY KEY,
                    qType VARCHAR(255),
                    question TEXT,
                    option1 VARCHAR(255),
                    option2 VARCHAR(255),
                    option3 VARCHAR(255),
                    option4 VARCHAR(255),
                    correct VARCHAR(255)
                );

            4) student_details
                CREATE TABLE student_details (
                    id INT,
                    firstname VARCHAR(50),
                    lastname VARCHAR(50),
                    middlename VARCHAR(50),
                    college VARCHAR(100),
                    rollno VARCHAR(20),
                    program VARCHAR(50),
                    stream VARCHAR(50),
                    year INT,
                    backlog INT,
                    currentcgpa DECIMAL(3,2),
                    email VARCHAR(100),
                    phoneno VARCHAR(15),
                    gender VARCHAR(10),
                    dob DATE,
                    nationality VARCHAR(50),
                    address VARCHAR(255),
                    FOREIGN KEY (id) REFERENCES student_register(id));
