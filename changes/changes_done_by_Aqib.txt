change : SQL tables:
            1) student_register
                CREATE TABLE student_register (
                        id INT AUTO_INCREMENT PRIMARY KEY,
                        username VARCHAR(255),
                        email VARCHAR(255) UNIQUE,
                        password VARCHAR(255),
                        enrollment_num VARCHAR(255) UNIQUE,
                        college VARCHAR(255),
                        course VARCHAR(255),
                        year VARCHAR(4),
                        rollno INT
                    );

            2) resume 
                CREATE TABLE IF NOT EXISTS resume (
                    id INT AUTO_INCREMENT PRIMARY KEY,
                    filename VARCHAR(255),
                    pdf_data LONGBLOB
                )

            3) quiz_questions
                CREATE TABLE quiz_questions (
                    id INT AUTO_INCREMENT PRIMARY KEY,
                    qType VARCHAR(255),
                    question TEXT,
                    option1 VARCHAR(255),
                    option2 VARCHAR(255),
                    option3 VARCHAR(255),
                    option4 VARCHAR(255),
                    correct VARCHAR(255)
                );

            4) student_details
                CREATE TABLE student_details (
                    id INT,
                    firstname VARCHAR(50),
                    lastname VARCHAR(50),
                    middlename VARCHAR(50),
                    college VARCHAR(100),
                    rollno VARCHAR(20),
                    program VARCHAR(50),
                    stream VARCHAR(50),
                    year INT,
                    backlog INT,
                    currentcgpa DECIMAL(3,2),
                    email VARCHAR(100),
                    phoneno VARCHAR(15),
                    gender VARCHAR(10),
                    dob DATE,
                    nationality VARCHAR(50),
                    address VARCHAR(255),
                    FOREIGN KEY (id) REFERENCES student_register(id));

            5) student resume
                CREATE TABLE student_resume (
                    id INT PRIMARY KEY NOT NULL,
                    file_name VARCHAR(255) NOT NULL,
                    FOREIGN KEY (id) REFERENCES student_register(id));

            6) job posting
                CREATE TABLE job_postings (
                    id INT AUTO_INCREMENT PRIMARY KEY,
                    job_role VARCHAR(255) NOT NULL,
                    job_type VARCHAR(50) NOT NULL,
                    skills_required TEXT NOT NULL,
                    num_employees INT NOT NULL,
                    num_openings INT NOT NULL,
                    company_description TEXT NOT NULL,
                    responsibilities TEXT NOT NULL);

            7) company registration
                CREATE TABLE company_registration (
                    id INT AUTO_INCREMENT PRIMARY KEY,
                    username VARCHAR(255) NOT NULL,
                    password1 VARCHAR(255) NOT NULL,
                    password2 VARCHAR(255) NOT NULL,
                    company_name VARCHAR(255) NOT NULL,
                    registration_number VARCHAR(255) NOT NULL,
                    address TEXT NOT NULL,
                    phone_number VARCHAR(20) NOT NULL,
                    email VARCHAR(255) NOT NULL,
                    industry_type VARCHAR(50) NOT NULL,
                    company_description TEXT NOT NULL,
                    logo_upload_filename VARCHAR(255) NOT NULL,
                    company_size INT NOT NULL);
                    
            8) applied student
                CREATE TABLE applied_student (
                    student_id INT,
                    company_id INT,
                    job_id INT,
                    FOREIGN KEY (student_id) REFERENCES student_details(id),
                    FOREIGN KEY (company_id) REFERENCES company_registration(id),
                    FOREIGN KEY (job_id) REFERENCES job_postings(id),
                    PRIMARY KEY (student_id, company_id, job_id));
            9) interviews
                CREATE TABLE interviews (
                    interview_id INT AUTO_INCREMENT PRIMARY KEY,
                    student_id INT,
                    job_id INT,
                    date DATE,
                    time TIME,
                    location VARCHAR(255),
                    UNIQUE (interview_id),
                    FOREIGN KEY (student_id) REFERENCES student_details(id),
                    FOREIGN KEY (job_id) REFERENCES job_postings(id));

            10) admin
                CREATE TABLE admin (
                    admin_id INT PRIMARY KEY AUTO_INCREMENT,
                    admin_name VARCHAR(255) NOT NULL,
                    admin_email VARCHAR(255) NOT NULL UNIQUE,
                    password VARCHAR(255) NOT NULL,
                    college_name VARCHAR(255) NOT NULL,
                    college_address VARCHAR(255) NOT NULL,
                    college_id INT UNIQUE);
